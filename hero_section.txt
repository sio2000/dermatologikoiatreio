    surfaceColor = mix(surfaceColor, highlightColor, refinement * 0.3);

    // Soft vignette
    float vignette = 1.0 - smoothstep(0.3, 1.2, dist);
    surfaceColor *= (0.7 + vignette * 0.3);

    float alpha = 0.92 + fresnel * 0.08;
    gl_FragColor = vec4(surfaceColor, alpha);
  }
`;

// Create liquid surface
const surfaceGeom = new THREE.PlaneGeometry(8, 8, 200, 200);
surfaceGeom.rotateX(-Math.PI / 2);
const surfaceMat = new THREE.ShaderMaterial({
  vertexShader: surfaceVertexShader,
  fragmentShader: surfaceFragmentShader,
  uniforms: {
    uTime: { value: 0 },
    uScroll: { value: 0 },
    uMouse: { value: new THREE.Vector2(0, 0) }
  },
  transparent: true,
  side: THREE.DoubleSide
});
const surface = new THREE.Mesh(surfaceGeom, surfaceMat);
surface.name = 'liquidSurface';
scene.add(surface);

// Drop light sphere
const dropGeom = new THREE.SphereGeometry(0.06, 32, 32);
const dropMat = new THREE.MeshBasicMaterial({ color: 0xffecd2, transparent: true, opacity: 0.9 });
const drop = new THREE.Mesh(dropGeom, dropMat);
drop.name = 'lightDrop';
drop.position.set(0, 2, 0);
scene.add(drop);

const dropLight = new THREE.PointLight(0xffd4a8, 1, 5);
dropLight.name = 'dropPointLight';
drop.add(dropLight);

// Post processing
const composer = new EffectComposer(renderer);
const renderPass = new RenderPass(scene, camera);
composer.addPass(renderPass);

const bloomPass = new UnrealBloomPass(
  new THREE.Vector2(window.innerWidth, window.innerHeight),
  0.6, 0.6, 0.7
);
composer.addPass(bloomPass);

const outputPass = new OutputPass();
composer.addPass(outputPass);

// Animation
const clock = new THREE.Clock();
let cameraStartZ = 4.5;

function animateHero() {
  const elapsed = clock.getElapsedTime();

  // Smooth mouse
  mouseX += (targetMouseX - mouseX) * 0.05;
  mouseY += (targetMouseY - mouseY) * 0.05;

  // Update uniforms
  surfaceMat.uniforms.uTime.value = elapsed;
  surfaceMat.uniforms.uScroll.value = scrollProgress;
  surfaceMat.uniforms.uMouse.value.set(mouseX, mouseY);

  // Drop animation (falls with scroll)
  const dropY = THREE.MathUtils.lerp(2.0, 0.05, Math.min(scrollProgress * 2, 1));
  drop.position.y = dropY;
  dropMat.opacity = scrollProgress < 0.5 ? 0.9 : THREE.MathUtils.lerp(0.9, 0, (scrollProgress - 0.5) * 2);
  dropLight.intensity = 1 + scrollProgress * 2;

  // Camera cinematic push-in
  camera.position.z = THREE.MathUtils.lerp(cameraStartZ, 3.2, scrollProgress * 0.5);
  camera.position.y = THREE.MathUtils.lerp(2.8, 2.2, scrollProgress * 0.3);
  camera.position.x = Math.sin(elapsed * 0.1) * 0.15 + mouseX * 0.3;
  camera.lookAt(0, 0, 0);

  // Bloom increases with scroll
  bloomPass.strength = 0.4 + scrollProgress * 0.8;

  composer.render();
}

renderer.setAnimationLoop(animateHero);

const finalCanvas = document.getElementById('finalCanvas');

let fW, fH;

function resizeFinalCanvas() {
  const rect = finalCanvas.parentElement.getBoundingClientRect();
  fW = finalCanvas.width = rect.width * Math.min(window.devicePixelRatio, 2);
  fH = finalCanvas.height = rect.height * Math.min(window.devicePixelRatio, 2);
  finalCanvas.style.width = rect.width + 'px';
  finalCanvas.style.height = rect.height + 'px';
}
resizeFinalCanvas();
window.addEventListener('resize', resizeFinalCanvas);

function drawFinalCanvas(time) {
  fCtx.fillStyle = '#0d0a08';
  fCtx.fillRect(0, 0, fW, fH);

  const cx = fW / 2, cy = fH / 2;
  const maxR = Math.max(fW, fH) * 0.5;

  // Soft glowing orb
  const pulse = 0.7 + Math.sin(time * 0.001) * 0.1;
  const grad = fCtx.createRadialGradient(cx, cy, 0, cx, cy, maxR * pulse);
  grad.addColorStop(0, 'rgba(255,228,190,0.12)');
  grad.addColorStop(0.3, 'rgba(200,160,110,0.06)');
  grad.addColorStop(0.6, 'rgba(100,70,40,0.03)');
  grad.addColorStop(1, 'rgba(13,10,8,0)');
  fCtx.fillStyle = grad;
  fCtx.fillRect(0, 0, fW, fH);

  // Secondary warm glow
  const grad2 = fCtx.createRadialGradient(cx + Math.sin(time * 0.0005) * 50, cy, 0, cx, cy, maxR * 0.4);
  grad2.addColorStop(0, 'rgba(255,200,150,0.06)');
  grad2.addColorStop(1, 'rgba(13,10,8,0)');
  fCtx.fillStyle = grad2;
  fCtx.fillRect(0, 0, fW, fH);

  requestAnimationFrame(drawFinalCanvas);
}
requestAnimationFrame(drawFinalCanvas);

function changeStep(dir) {
  const newStep = currentStep + dir;
  if (newStep < 0 || newStep >= totalSteps) return;

  if (newStep === totalSteps - 1) {
    document.getElementById('bookingNext').textContent = 'Επιβεβαίωση';
  } else {
    document.getElementById('bookingNext').innerHTML = 'Επόμενο <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M5 12h14M12 5l7 7-7 7"/></svg>';
  }

  document.querySelectorAll('.booking-step').forEach((s, i) => s.classList.toggle('active', i === currentStep));
  document.querySelectorAll('.booking-panel').forEach((p, i) => p.classList.toggle('active', i === currentStep));
  document.getElementById('bookingPrev').style.visibility = currentStep === 0 ? 'hidden' : 'visible';
}
