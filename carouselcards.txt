const surfaceGeo = new THREE.PlaneGeometry(6, 6, 200, 200);

const surfaceVertexShader = `
  uniform float uTime;
  uniform float uState;
  uniform float uMouse;
  varying vec2 vUv;
  varying float vElevation;
  varying vec3 vNormal;
  varying vec3 vPosition;

  // Simplex noise
  vec3 mod289(vec3 x){return x - floor(x * (1.0 / 289.0)) * 289.0;}
  vec4 mod289(vec4 x){return x - floor(x * (1.0 / 289.0)) * 289.0;}
  vec4 permute(vec4 x){return mod289(((x*34.0)+1.0)*x);}
  vec4 taylorInvSqrt(vec4 r){return 1.79284291400159 - 0.85373472095314 * r;}
  float snoise(vec3 v){
    const vec2 C = vec2(1.0/6.0, 1.0/3.0);
    const vec4 D = vec4(0.0, 0.5, 1.0, 2.0);
    vec3 i  = floor(v + dot(v, C.yyy));
    vec3 x0 = v - i + dot(i, C.xxx);
    vec3 g = step(x0.yzx, x0.xyz);
    vec3 l = 1.0 - g;
    vec3 i1 = min(g.xyz, l.zxy);
    vec3 i2 = max(g.xyz, l.zxy);
    vec3 x1 = x0 - i1 + C.xxx;
    vec3 x2 = x0 - i2 + C.yyy;
    vec3 x3 = x0 - D.yyy;
    i = mod289(i);
    vec4 p = permute(permute(permute(
              i.z + vec4(0.0, i1.z, i2.z, 1.0))
            + i.y + vec4(0.0, i1.y, i2.y, 1.0))
            + i.x + vec4(0.0, i1.x, i2.x, 1.0));
    float n_ = 0.142857142857;
    vec3 ns = n_ * D.wyz - D.xzx;
    vec4 j = p - 49.0 * floor(p * ns.z * ns.z);
    vec4 x_ = floor(j * ns.z);
    vec4 y_ = floor(j - 7.0 * x_);
    vec4 x = x_ * ns.x + ns.yyyy;
    vec4 y = y_ * ns.x + ns.yyyy;
    vec4 h = 1.0 - abs(x) - abs(y);
    vec4 b0 = vec4(x.xy, y.xy);
    vec4 b1 = vec4(x.zw, y.zw);
    vec4 s0 = floor(b0)*2.0 + 1.0;
    vec4 s1 = floor(b1)*2.0 + 1.0;
    vec4 sh = -step(h, vec4(0.0));
    vec4 a0 = b0.xzyw + s0.xzyw*sh.xxyy;
    vec4 a1 = b1.xzyw + s1.xzyw*sh.zzww;
    vec3 p0 = vec3(a0.xy,h.x);
    vec3 p1 = vec3(a0.zw,h.y);
    vec3 p2 = vec3(a1.xy,h.z);
    vec3 p3 = vec3(a1.zw,h.w);
    vec4 norm = taylorInvSqrt(vec4(dot(p0,p0), dot(p1,p1), dot(p2,p2), dot(p3,p3)));
    p0 *= norm.x; p1 *= norm.y; p2 *= norm.z; p3 *= norm.w;
    vec4 m = max(0.6 - vec4(dot(x0,x0), dot(x1,x1), dot(x2,x2), dot(x3,x3)), 0.0);
    m = m * m;
    return 42.0 * dot(m*m, vec4(dot(p0,x0), dot(p1,x1), dot(p2,x2), dot(p3,x3)));
  }

  void main(){
    vUv = uv;
    vec3 pos = position;

    float baseWave = snoise(vec3(pos.xy * 1.5, uTime * 0.15)) * 0.15;
    float detailWave = snoise(vec3(pos.xy * 3.0, uTime * 0.25)) * 0.05;

    // State-based displacement
    float analysisGrid = 0.0;
    if(uState > 0.5 && uState < 1.5){

    }

    float treatmentSmooth = 0.0;
    if(uState > 1.5 && uState < 2.5){
      treatmentSmooth = snoise(vec3(pos.xy * 5.0, uTime * 0.5)) * 0.08 * (uState - 1.5);
    }

    float resultCalm = 0.0;
    if(uState > 2.5){
      resultCalm = snoise(vec3(pos.xy * 0.8, uTime * 0.1)) * 0.1 * min(uState - 2.5, 1.0);
    }

    float elevation = baseWave + detailWave + analysisGrid + treatmentSmooth + resultCalm;
    pos.z += elevation;

    vElevation = elevation;
    vNormal = normalize(normal + vec3(elevation * 2.0));
    vPosition = pos;

    gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
  }
`;

const surfaceFragmentShader = `
  uniform float uTime;
  uniform float uState;
  uniform vec2 uMousePos;
  varying vec2 vUv;
  varying float vElevation;
  varying vec3 vNormal;
  varying vec3 vPosition;

  void main(){
    // Base color: warm champagne tones
    vec3 baseColor = vec3(0.85, 0.78, 0.68);
    vec3 deepColor = vec3(0.15, 0.12, 0.08);
    vec3 accentColor = vec3(0.79, 0.66, 0.49);
    vec3 glowColor = vec3(1.0, 0.92, 0.78);

    // Fresnel-like edge glow
    vec3 viewDir = normalize(cameraPosition - vPosition);
    float fresnel = pow(1.0 - max(dot(vNormal, viewDir), 0.0), 3.0);

    // Base mix
    vec3 color = mix(deepColor, baseColor, vUv.y * 0.6 + 0.2);

    // Elevation coloring
    color = mix(color, accentColor, vElevation * 3.0 + 0.3);

    // Mouse light influence
    float mouseDist = distance(vUv, uMousePos);
    float mouseLight = smoothstep(0.4, 0.0, mouseDist) * 0.6;
    color += glowColor * mouseLight;

    // State-based effects
    // Analysis: scanning lines
    if(uState > 0.5 && uState < 2.0){
      float scanLine = smoothstep(0.01, 0.0, abs(sin(vUv.y * 40.0 + uTime * 3.0) - 0.95));
      color += vec3(0.3, 0.5, 0.7) * scanLine * min(uState - 0.5, 1.0) * 0.5;

      float gridX = smoothstep(0.02, 0.0, abs(fract(vUv.x * 15.0) - 0.5) - 0.48);
      float gridY = smoothstep(0.02, 0.0, abs(fract(vUv.y * 15.0) - 0.5) - 0.48);
      color += vec3(0.2, 0.4, 0.6) * (gridX + gridY) * min(uState - 0.5, 1.0) * 0.15;
    }

    // Treatment: warm glow
    if(uState > 1.5 && uState < 3.0){
      float treatGlow = sin(uTime * 2.0) * 0.5 + 0.5;
      color += glowColor * treatGlow * 0.2 * min(uState - 1.5, 1.0);
      float beam = smoothstep(0.15, 0.0, abs(vUv.x - 0.5 + sin(uTime) * 0.2));
      color += vec3(1.0, 0.85, 0.6) * beam * 0.3 * min(uState - 1.5, 1.0);
    }

    // Result: radiance
    if(uState > 2.5){
      float radiance = smoothstep(0.5, 0.0, distance(vUv, vec2(0.5))) * 0.3;
      color += glowColor * radiance * min(uState - 2.5, 1.0);
      color = mix(color, glowColor * 0.9, fresnel * 0.4 * min(uState - 2.5, 1.0));
    }

    // Fresnel edge
    color += accentColor * fresnel * 0.3;

    // Alpha based on distance from center
    float alpha = smoothstep(3.5, 1.5, length(vPosition.xy));

    gl_FragColor = vec4(color, alpha);
  }
`;

const clock = new THREE.Clock();

function animate() {
  const elapsed = clock.getElapsedTime();
  const delta = clock.getDelta();

  // Smooth state transition
  currentState += (targetState - currentState) * 0.03;

  // Update uniforms
  surfaceMaterial.uniforms.uTime.value = elapsed;
  surfaceMaterial.uniforms.uState.value = currentState;
  surfaceMaterial.uniforms.uMousePos.value.set(mouse.x, mouse.y);

  particleMat.uniforms.uTime.value = elapsed;
  particleMat.uniforms.uState.value = currentState;

  lines.forEach((line) => {
    line.material.uniforms.uTime.value = elapsed;
    line.material.uniforms.uState.value = currentState;
  });

  // Camera subtle motion
  camera.position.x += (mouse.x * 0.5 - 0.25 - camera.position.x) * 0.02;
  camera.position.y += (mouse.y * 0.3 - 0.15 - camera.position.y) * 0.02;
  camera.lookAt(0, 0, 0);

  // Bloom intensity based on state
  bloomPass.strength = 0.6 + currentState * 0.15;

  composer.render();
}

renderer.setAnimationLoop(animate);

// ============ RESIZE ============
window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
  composer.setSize(window.innerWidth, window.innerHeight);
});

let carouselAngle = 0;
let carouselAutoSpeed = 0.002;
let carouselDragSpeed = 0;
let isDragging = false;
let lastX = 0;

const carouselContainer = document.getElementById('carousel-container');
carouselContainer.addEventListener('mousedown', (e) => { isDragging = true; lastX = e.clientX; });
window.addEventListener('mouseup', () => { isDragging = false; });
window.addEventListener('mousemove', (e) => {
  if (isDragging) {
    carouselDragSpeed = (e.clientX - lastX) * 0.003;
    lastX = e.clientX;
  }
});

function updateCarousel() {
  if (!isDragging) {
    carouselDragSpeed *= 0.95;
  }
  carouselAngle += carouselAutoSpeed + carouselDragSpeed;

  carouselCards.forEach((card, i) => {
    const angle = carouselAngle + i * cardAngle;
    const x = Math.sin(angle) * radius;
    const z = Math.cos(angle) * radius - radius;
    const scale = (z + radius * 2) / (radius * 2);
    const opacity = 0.3 + scale * 0.7;
    const rotY = -angle * (180 / Math.PI) * 0.3;

    card.style.transform = `translateX(${x}px) translateZ(${z}px) scale(${0.6 + scale * 0.4}) rotateY(${rotY}deg)`;
    card.style.opacity = opacity;
    card.style.zIndex = Math.round(scale * 100);
  });

  requestAnimationFrame(updateCarousel);
}
updateCarousel();

  neuralSvg.appendChild(path);

    startX: Math.random() * window.innerWidth,
    startY: Math.random() * window.innerHeight,
    endX: Math.random() * window.innerWidth,
    endY: Math.random() * window.innerHeight,
    speed: Math.random() * 0.3 + 0.1,
    offset: Math.random() * Math.PI * 2,
  });
}

function updateNeuralLines(time) {
  neuralPaths.forEach(p => {
    const mx = Math.sin(time * p.speed + p.offset) * 100;
    const my = Math.cos(time * p.speed * 0.7 + p.offset) * 80;
    const cx1 = (p.startX + p.endX) / 2 + mx;
    const cy1 = (p.startY + p.endY) / 2 + my;
    p.el.setAttribute('d', `M${p.startX},${p.startY} Q${cx1},${cy1} ${p.endX},${p.endY}`);
  });
  requestAnimationFrame(() => updateNeuralLines(performance.now() / 1000));
}
